Karazana Data Types

Solidity dia fiteny fandaharana statically typed, izany hoe mila manondro ny karazana data isehoantsika isaky ny mamorona variable. Ireto ny karazana fototra:

String

Ho an'ny lahatsoratra

Boolean

True na false ihany

Integers

Isa - misy signed (int) sy unsigned (uint)

Address

Adiresy Ethereum wallet

Ohatra fampiharana

Ity misy ohatra smart contract tsotra izay mampiasa data types samihafa:

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.30;

contract UserProfile {
    // State variables - mikarakara data eo amin'ny blockchain
    string public firstName = "Rabe";
    string public lastName = "Andry";
    uint256 public age = 30;
    bool public isVerified = false;
    address public walletAddress;
    
    // Constructor - miasa rehefa deploy ny contract
    constructor() {
        walletAddress = msg.sender; // ny olona mandefa ny contract
    }
    
    // Function hanovana ny anarana
    function updateName(string memory _firstName, string memory _lastName) public {
        firstName = _firstName;
        lastName = _lastName;
    }
    
    // Function hanamarina ny account
    function verifyAccount() public {
        require(msg.sender == walletAddress, "Tsy izy ny tompony!");
        isVerified = true;
    }
}

    }
}

Fampiharana: Product Registry

Ity misy ohatra fampiharana mitahiry sy mifehy products:

contract ProductRegistry {
    struct Product {
        uint256 id;
        string name;
        uint256 price;  // amin'ny wei
        address seller;
        bool isAvailable;
    }
    
    Product[] public products;
    mapping(uint256 => Product) public productById;
    mapping(address => uint256[]) public productsBySeller;
    
    uint256 public nextProductId = 1;
    
    function addProduct(string memory _name, uint256 _price) public {
        Product memory newProduct = Product({
            id: nextProductId,
            name: _name,
            price: _price,
            seller: msg.sender,
            isAvailable: true
        });
        
        products.push(newProduct);
        productById[nextProductId] = newProduct;
        productsBySeller[msg.sender].push(nextProductId);
        
        nextProductId++;
    }
    
    function buyProduct(uint256 _productId) public payable {
        Product storage product = productById[_productId];
        
        require(product.isAvailable, "Tsy amidy intsony!");
        require(msg.value >= product.price, "Tsy ampy ny vola!");
        
        product.isAvailable = false;
        
        // Mandefa vola any amin'ny mpivarotra
        payable(product.seller).transfer(product.price);
        
        // Mamerina ny ambiny (raha misy)
        if (msg.value > product.price) {
            payable(msg.sender).transfer(msg.value - product.price);
        }
    }
}

5. Functions sy Modifiers

Function Visibility

Amin'ny Solidity dia misy karazana visibility efa-misoratra ho an'ny functions. Isaky ny mamorona function dia tsy maintsy manondro izay:

public

Azo antsoina na avy aiza na avy aiza (anaty contract na avy ivelany)

external

Azo antsoina avy ivelany ihany (tsy avy anaty contract)

internal

Azo ampiasaina anaty contract sy contract manaraka azy ihany

private

Anaty contract iray ihany no azo ampiasaina

View sy Pure Functions

  • • view: Mamaky state variables nefa tsy manova azy
  • • pure: Tsy mamaky na manova state variables akory
contract FunctionExamples {
    uint256 public count = 0;
    
    // Public function - azo antsoina na avy aiza
    function increment() public {
        count++;
    }
    
    // View function - mamaky ihany
    function getCount() public view returns (uint256) {
        return count;
    }
    
    // Pure function - calculation ihany
    function add(uint256 a, uint256 b) public pure returns (uint256) {
        return a + b;
    }
    
    // Private function - anaty contract ihany
    function _validateInput(uint256 _input) private pure returns (bool) {
        return _input > 0;
    }
}

Modifiers - Fitsapana Access Control

Modifiers dia ahafahantsika manao fitsapana mialoha ny function. Matetika ampiasaina amin'ny access control, ohatra ny "owner ihany no afaka". 

contract AccessControl {
    address public owner;
    mapping(address => bool) public admins;
    
    constructor() {
        owner = msg.sender;
    }
    
    // Modifier fitsapana raha owner
    modifier onlyOwner() {
        require(msg.sender == owner, "Tsy izy ny owner!");
        _; // eto no hitizan'ny function content
    }
    
    // Modifier fitsapana raha admin
    modifier onlyAdmin() {
        require(admins[msg.sender] || msg.sender == owner, "Tsy admin!");
        _;
    }
    
    // Function hamadihana owner (owner ihany no afaka)
    function transferOwnership(address _newOwner) public onlyOwner {
        require(_newOwner != address(0), "Address tsy mety!");
        owner = _newOwner;
    }
    
    // Function hanendry admin
    function addAdmin(address _admin) public onlyOwner {
        admins[_admin] = true;
    }
    
    // Function hanesorana admin
    function removeAdmin(address _admin) public onlyOwner {
        admins[_admin] = false;
    }
    
    // Function izay admin ihany no afaka
    function adminOnlyFunction() public onlyAdmin {
        // Function content eto
    }
}

Fampiharana: Voting System

Ity misy ohatra voting system tsotra izay owner ihany no afaka mi-reset votes:

contract VotingSystem {
    address public owner;
    string public proposal;
    uint256 public yesVotes;
    uint256 public noVotes;
    bool public votingActive;
    
    mapping(address => bool) public hasVoted;
    
    constructor(string memory _proposal) {
        owner = msg.sender;
        proposal = _proposal;
        votingActive = true;
    }
    
    modifier onlyOwner() {
        require(msg.sender == owner, "Tsy izy ny owner!");
        _;
    }
    
    modifier votingIsActive() {
        require(votingActive, "Tapitra ny voting!");
        _;
    }
    
    modifier hasNotVoted() {
        require(!hasVoted[msg.sender], "Efa nifidy!");
        _;
    }
    
    function voteYes() public votingIsActive hasNotVoted {
        yesVotes++;
        hasVoted[msg.sender] = true;
    }
    
    function voteNo() public votingIsActive hasNotVoted {
        noVotes++;
        hasVoted[msg.sender] = true;
    }
    
    function endVoting() public onlyOwner {
        votingActive = false;
    }
    
    function resetVotes() public onlyOwner {
        yesVotes = 0;
        noVotes = 0;
        votingActive = true;
        
        // Reset hasVoted mapping dia sarotra, mila liste ny voters
        // Ohatra tsotra ihany ity
    }
    
    function getResults() public view returns (uint256, uint256) {
        return (yesVotes, noVotes);
    }
}

SimpleDAO Contract

Ity misy ohatra DAO tsotra izay ahafahana mamorona proposal sy mifidy:

// SPDX-License-Identifier: MIT
pragma solidity ^0.8.30;

contract SimpleDAO {
    // Enum ho an'ny status proposal
    enum ProposalStatus { 
        Pending,    // Mbola miandry
        Approved,   // Nekena
        Rejected,   // Lavina
        Executed    // Vita
    }
    
    struct Proposal {
        uint256 id;
        string description;
        address proposer;
        uint256 yesVotes;
        uint256 noVotes;
        ProposalStatus status;
        uint256 deadline;
        uint256 amount;  // vola takiana (raha misy)
        address recipient; // izay hahazo ny vola
    }
    
    address public owner;
    uint256 public proposalCount;
    uint256 public memberCount;
    uint256 public constant VOTING_PERIOD = 7 days;
    uint256 public constant MIN_VOTES_TO_PASS = 3;
    
    mapping(uint256 => Proposal) public proposals;
    mapping(address => bool) public members;
    mapping(uint256 => mapping(address => bool)) public hasVoted;
    
    event ProposalCreated(uint256 indexed proposalId, string description, address indexed proposer);
    event VoteCast(uint256 indexed proposalId, address indexed voter, bool support);
    event ProposalExecuted(uint256 indexed proposalId);
    
    constructor() {
        owner = msg.sender;
        members[msg.sender] = true;
        memberCount = 1;
    }
    
    modifier onlyMember() {
        require(members[msg.sender], "Tsy mpikambana!");
        _;
    }
    
    modifier onlyOwner() {
        require(msg.sender == owner, "Tsy owner!");
        _;
    }
    
    // Hanendry mpikambana vaovao
    function addMember(address _newMember) public onlyOwner {
        require(!members[_newMember], "Efa mpikambana sahady!");
        require(_newMember != address(0), "Address tsy mety!");
        
        members[_newMember] = true;
        memberCount++;
    }
    
    // Hamoronana proposal
    function createProposal(
        string memory _description,
        uint256 _amount,
        address _recipient
    ) public onlyMember {
        proposalCount++;
        
        proposals[proposalCount] = Proposal({
            id: proposalCount,
            description: _description,
            proposer: msg.sender,
            yesVotes: 0,
            noVotes: 0,
            status: ProposalStatus.Pending,
            deadline: block.timestamp + VOTING_PERIOD,
            amount: _amount,
            recipient: _recipient
        });
        
        emit ProposalCreated(proposalCount, _description, msg.sender);
    }
    
    // Fifidianana
    function vote(uint256 _proposalId, bool _support) public onlyMember {
        Proposal storage proposal = proposals[_proposalId];
        
        require(proposal.id != 0, "Proposal tsy misy!");
        require(proposal.status == ProposalStatus.Pending, "Tsy azo fidiana intsony!");
        require(block.timestamp <= proposal.deadline, "Tapitra ny deadline!");
        require(!hasVoted[_proposalId][msg.sender], "Efa nifidy!");
        
        hasVoted[_proposalId][msg.sender] = true;
        
        if (_support) {
            proposal.yesVotes++;
        } else {
            proposal.noVotes++;
        }
        
        emit VoteCast(_proposalId, msg.sender, _support);
        
        // Jereo raha tokony hiova ny status
        _updateProposalStatus(_proposalId);
    }
    
    // Function hanavao ny status proposal
    function _updateProposalStatus(uint256 _proposalId) internal {
        Proposal storage proposal = proposals[_proposalId];
        
        uint256 totalVotes = proposal.yesVotes + proposal.noVotes;
        
        if (totalVotes >= MIN_VOTES_TO_PASS) {
            if (proposal.yesVotes > proposal.noVotes) {
                proposal.status = ProposalStatus.Approved;
            } else {
                proposal.status = ProposalStatus.Rejected;
            }
        } else if (block.timestamp > proposal.deadline) {
            proposal.status = ProposalStatus.Rejected;
        }
    }
    
    // Hanatanterahana proposal nekena
    function executeProposal(uint256 _proposalId) public onlyMember {
        Proposal storage proposal = proposals[_proposalId];
        
        require(proposal.status == ProposalStatus.Approved, "Tsy nekena ny proposal!");
        require(address(this).balance >= proposal.amount, "Tsy ampy ny vola!");
        
        proposal.status = ProposalStatus.Executed;
        
        if (proposal.amount > 0 && proposal.recipient != address(0)) {
            payable(proposal.recipient).transfer(proposal.amount);
        }
        
        emit ProposalExecuted(_proposalId);
    }
    
    // Mandefa vola ho any amin'ny DAO
    function contribute() public payable {
        require(msg.value > 0, "Tsy misy vola!");
    }
    
    // Jereo ny vola ananan'ny DAO
    function getBalance() public view returns (uint256) {
        return address(this).balance;
    }
    
    // Jereo info momba ny proposal
    function getProposal(uint256 _proposalId) public view returns (
        string memory description,
        address proposer,
        uint256 yesVotes,
        uint256 noVotes,
        ProposalStatus status,
        uint256 deadline
    ) {
        Proposal storage proposal = proposals[_proposalId];
        return (
            proposal.description,
            proposal.proposer,
            proposal.yesVotes,
            proposal.noVotes,
            proposal.status,
            proposal.deadline
        );
    }
}

7. Fiarovana sy Best Practices

Fahadisoana mahazatra sy fomba fiarovana

Smart contract development dia mila fitandremana be satria rehefa deploy dia sarotra ovaina. Ireto ny fahadisoana mahazatra sy ny fomba fisorohana azy:

1. Reentrancy Attacks

Reentrancy dia fanafihana izay mampiasa ny fahadisoana amin'ny fandrefesana external calls. Ny manam-pahefana dia afaka miantso ny function indray mandeha alohan'ny hihitry ny transaction voalohany.

❌ TSY METY
function withdraw() public {
    uint256 amount = balances[msg.sender];
    (bool success,) = msg.sender.call{value: amount}("");
    require(success);
    balances[msg.sender] = 0; // LATE!
}
✅ METY
function withdraw() public {
    uint256 amount = balances[msg.sender];
    balances[msg.sender] = 0; // EARLY!
    (bool success,) = msg.sender.call{value: amount}("");
    require(success);
}

2. Integer Overflow/Underflow

Alohan'ny Solidity 0.8.0 dia tsy nisy automatic checking ho an'ny overflow. Ankehitriny dia misy sahady, fa mbola tsara raha manampy fitsapana. 

// Solidity 0.8+ - automatic overflow protection
function safeAdd(uint256 a, uint256 b) public pure returns (uint256) {
    return a + b; // Auto-revert rehefa overflow
}

// Manual checking (raha ilaina)
function manualAdd(uint256 a, uint256 b) public pure returns (uint256) {
    require(a + b >= a, "Overflow!");
    return a + b;
}

3. Access Control tsara

Ampiasa OpenZeppelin na manao modifiers tsara ho an'ny access control. Aza miantehitra amin'ny tx.origin. 

// Import OpenZeppelin
import "@openzeppelin/contracts/access/Ownable.sol";
import "@openzeppelin/contracts/security/ReentrancyGuard.sol";

contract SecureContract is Ownable, ReentrancyGuard {
    mapping(address => uint256) public balances;
    
    function withdraw() public nonReentrant {
        uint256 amount = balances[msg.sender];
        require(amount > 0, "Tsy misy balance!");
        
        balances[msg.sender] = 0;
        (bool success,) = msg.sender.call{value: amount}("");
        require(success, "Transfer tsy nahomby!");
    }
}

Nisy fanatsarana be dia be amin'ny security features sy error handling. Ny compiler dia mahatonga azy ireo mitaky fandikana mivaingana kokoa. 

// Solidity 0.8.30 example
pragma solidity ^0.8.30;

contract Modern2024Contract {
    // Vaovao: Custom errors with parameters
    error InsufficientBalance(uint256 available, uint256 required);
    error UnauthorizedAccess(address caller);
    
    mapping(address => uint256) private balances;
    
    function withdraw(uint256 amount) public {
        uint256 balance = balances[msg.sender];
        
        // Using new custom error format
        if (balance < amount) {
            revert InsufficientBalance(balance, amount);
        }
        
        balances[msg.sender] -= amount;
        payable(msg.sender).transfer(amount);
    }
}
